export async function getJourneyData(userId: string, now: Date = new Date()): Promise<JourneyData> {
  // 1. Resolve player via player_members where user_id = userId
  // 2. If no player or no Supabase -> { player:null, missions:[], empty:true, levelLabel:"" }
  // 3. Fetch event_id via player.cohort_id -> cohorts.event_id
  // 4. Fetch missions where event_id = event_id, ordered by scheduled_at asc, level_id asc, ord asc
  // 5. Filter missions du JOUR : same calendar date as `now` OR scheduled_at IS NULL

  // 7. Fetch submissions where player_id = player.id and deliverable_template_id in (...)
  // 8. Group by mission, compute status for each deliverable using computeDeliverableStatus
  // 9. Mission status: passe if scheduled_at + 1h < now; en_cours if scheduled_at <= now <= scheduled_at + 1h; sinon a_venir

export function computeDeliverableStatus(submissions: { version: number; status: SubmissionStatus }[]): { status: DeliverableStatus; latestSubmissionId: string | null } {
  if (submissions.length === 0) return { status: "a_rendre", latestSubmissionId: null };
  // pick max(version), return its status
}

export function missionStatus(scheduledAt: string | null, now: Date): MissionStatus {
  if (!scheduledAt) return "a_venir";
  const t = new Date(scheduledAt).getTime();
  const n = now.getTime();
  if (n < t) return "a_venir";
  if (n <= t + 60 * 60 * 1000) return "en_cours";
  return "passe";
}
