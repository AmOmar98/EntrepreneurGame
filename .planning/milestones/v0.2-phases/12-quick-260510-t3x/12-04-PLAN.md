```powershell
# 1. Verifier toolchain Supabase CLI
npx supabase --version
# Attendu : >= 1.x

# 2. Verifier lien projet
npx supabase status
# Attendu : project ref affiche, no error

# 3. Lister migrations en attente

# Attendu : 20260510170000_bonus_events_recreate et 20260510170100_moscow_cards

# 5. Verifier presence tables
psql $env:SUPABASE_DB_URL -c "\dt public.bonus_events public.moscow_cards"

# 7. Verifier RLS active
psql $env:SUPABASE_DB_URL -c "select tablename, rowsecurity from pg_tables where tablename in ('bonus_events','moscow_cards')"
# Attendu : 2 lignes, rowsecurity = t pour les deux

# 8. Verifier policies
psql $env:SUPABASE_DB_URL -c "select tablename, policyname from pg_policies where tablename in ('bonus_events','moscow_cards') order by tablename, policyname"
# Attendu : 8 lignes minimum (4 par table : select, *_insert, *_update, *_delete)

# 9. Idempotency re-run
npx supabase migration up --linked
# Attendu : no-op, "Remote database is up to date" ou equivalent

# 10. (Optionnel) Re-apply mirror local pour confirmer idempotency au niveau psql

# Attendu : 0 erreur, 0 ligne inseree (DO blocks + IF NOT EXISTS + DROP+CREATE policies)
```
