```sql
-- Fonction recalcul agrégé (idempotente, security definer)
create or replace function public.recalc_player_engagement(p_player_id uuid)
returns void
language plpgsql
security definer
set search_path = public
-- Cumulatif par (player, deliverable_template) :
--   +100 si ≥1 submission existe (palier "Soumis", irréversible)
--   +25  si ≥1 evaluation existe (palier "Reviewed", irréversible)
--   +50  si verdict le plus récent ∈ {validate_v1, validate_v2} (recalculé, Q5=A)
-- Total max par livrable validé = 175 pts.

-- Trigger 1 : after insert/update/delete on public.submissions
create trigger trg_submission_engagement
  after insert or update or delete on public.submissions
  for each row execute function public.on_submission_engagement_change();

-- Trigger 2 : after insert/update/delete on public.evaluations
create trigger trg_evaluation_engagement
  after insert or update or delete on public.evaluations
  for each row execute function public.on_evaluation_engagement_change();
```
